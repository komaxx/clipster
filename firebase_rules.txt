# Rules required to allow each user access to their clipz - and only theirs.
# Copy the lines below into your Firebase/Database(firestore)/Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
  	match /clipz/{doc} {
    	allow read, write: 
      if request.auth != null 
      && request.auth.uid != null
      && request.auth.uid == doc;
    }
  
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
